//This program demonstrates block scoping
// A.A. Anda

#include <iostream>
using namespace std;

int x = 1; //global x

void foo(char);

int main()
{ //scope A
    int x = 2; //local to main. Hides global x
    cout << "x in main is " << x << endl;
    cout << "revealed global x in main is " << ::x << endl;
    foo('A');

    { //scope B
	int x = 3; //local to scope B. Hides x in main.
	cout << "x in scope B is " << x << endl;
	cout << "revealed global x in scope B is " << ::x << endl;
	foo('B');

	{ //scope C
	    int x = 4; //local to scope C. Hides x in scope B.
	    cout << "x in scope C is " << x << endl;
	    cout << "revealed global x in scope C is " << ::x << endl;
	    foo('C');

	    for(int x=5; x<7;x++)
	    { // scope D
		cout << "x in scope D is " << x << endl;
		cout << "revealed global x in scope D is " << ::x << endl;
		foo('D');
	    }

	    cout << "x in scope C is " << x << endl;
	    foo('C');

	}

	cout << "x in scope B is " << x << endl;
    }

    cout << "x in main is " << x << endl;
}

void foo(char y)
{
    cout << "x in foo in region " << y << " is " << x << endl << endl;
}
